      "leanquill.newCharacter",
      async () => {
        const name = await vscode.window.showInputBox({
          prompt: "Character name",
          placeHolder: "e.g. Jane Doe",
        });
        if (!name?.trim()) {

        }

          await planningPanelProvider.refresh();

    );

    **Part A: Add manuscript document event listeners to extension.ts**

    After the `newCharacterCommand` registration (Task 1), add:
    ```typescript
    // Manuscript save listener — trigger character mention scan (D-06)
    const onSaveListener = vscode.workspace.onDidSaveTextDocument(async (doc) => {
      // Only scan files inside the manuscript folder
      const rel = path.relative(rootPath, doc.uri.fsPath).replace(/\\/g, "/");
      if (!rel.startsWith("manuscript/") || !rel.endsWith(".md")) {
        return;
      }
      try {
        const latestConfig = await readProjectConfig(rootPath) ?? {
          schemaVersion: "1",
          folders: { research: "research/leanquill/", characters: "notes/characters/" },
        };
        await scanManuscriptFileForCharacters(doc.uri.fsPath, rootPath, latestConfig, safeFs);
        await planningPanelProvider.refresh();
      } catch (err: unknown) {
        // Silent — character scan errors must not interrupt the author's flow
        console.error("[LeanQuill] Character scan error on save:", err);
      }
    });

    // Manuscript open listener — trigger character mention scan on first open (D-06)
    const onOpenListener = vscode.workspace.onDidOpenTextDocument(async (doc) => {
      const rel = path.relative(rootPath, doc.uri.fsPath).replace(/\\/g, "/");
      if (!rel.startsWith("manuscript/") || !rel.endsWith(".md")) {
        return;
      }
      try {
        const latestConfig = await readProjectConfig(rootPath) ?? {
          schemaVersion: "1",
          folders: { research: "research/leanquill/", characters: "notes/characters/" },
        };
        await scanManuscriptFileForCharacters(doc.uri.fsPath, rootPath, latestConfig, safeFs);
        await planningPanelProvider.refresh();
      } catch (err: unknown) {
        console.error("[LeanQuill] Character scan error on open:", err);
      }
    });
    ```

    Push both to subscriptions: `context.subscriptions.push(onSaveListener, onOpenListener);`

    Note on `planningPanelProvider.refresh()`: `refresh()` is a no-op if the panel is not open (per Plan 02 — `_renderPanel` checks `if (!this._panel) return`). So calling it unconditionally is safe.
